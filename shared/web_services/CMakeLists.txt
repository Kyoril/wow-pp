# Collect source and header files
file(GLOB srcFiles "./*.cpp" "./*.h" "./*.hpp")
remove_pch_cpp(srcFiles "${CMAKE_CURRENT_SOURCE_DIR}/pch.cpp")

# Add source groups
source_group(src FILES ${srcFiles})
	
# Add library project
add_library(web_services ${srcFiles})
add_precompiled_header(web_services "${CMAKE_CURRENT_SOURCE_DIR}/pch.h")

# Solution folder
if(MSVC)
	set_property(TARGET web_services PROPERTY FOLDER "shared")
endif(MSVC)
